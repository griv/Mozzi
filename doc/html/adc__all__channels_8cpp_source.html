00071 /* This is called with when the adc finishes a conversion.
00072 It puts the new reading into the sensors array and starts a new reading on the next channel.
00073 */
00074 ISR(ADC_vect, ISR_BLOCK)
00075 {
00076                 static boolean secondRead = false;
00077                 //Only record the second read on each channel
00078                 if(secondRead){
00079                                 //sensors[current_adc] = ADCL | (ADCH << 8);
00080                                 //bobgardner: ..The compiler is clever enough to read the 10 bit value like this: val=ADC;
00081                                 sensors[current_adc] = ADC;
00082                                 current_adc++;
00083                                 if(current_adc > NUM_ANALOG_INPUTS){
00084                                                 //Sequence complete.  Stop A2D conversions
00085                                                 readComplete = true;
00086                                 }
00087                                 else{
00088                                                 //Switch to next channel
00089                                                 
00090                                                 //ADMUX = (1 << REFS0) | current_adc;
00091                                                 //ADCSRA |= (1 << ADSC);
00092                                                 
00093                                                 adcStartConversion(current_adc);
00094                                 }
00095                                 secondRead = false;
00096                 }
00097                 else{
00098                                 secondRead = true;
